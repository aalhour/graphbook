\dontprintsemicolon
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
\SetKwData{NULL}{NULL}
%%
%% input/output
\Input{A weighted connected graph $G = (V, E)$ with weight function $w$.}
\Output{A minimum spanning tree $T$ of $G$.}
\BlankLine
%%
%% algorithm body
\For{\rm each $v \in V$}{
  $\cost[v] \assign \infty$\;
  $\parent[v] \assign \NULL$\;
}
$r \assign$ aribtrary vertex of $V$\;
$\cost[r] \assign 0$\;
$Q \assign V$\;
\While{$Q \neq \emptyset$}{
  $u \assign \extractMin(Q)$\;
  \For{\rm each $v \in \adj(u)$}{
    \If{\rm $v \in Q$ and $w(u,v) < \cost[v]$}{
      $\parent[v] \assign u$\;
      $\cost[v] \assign w(u,v)$\;
    }
  }
}
$T \assign \big\{(v, \parent[v]) \mid v \in V - \{r\}\big\}$\;
\Return $T$\;
