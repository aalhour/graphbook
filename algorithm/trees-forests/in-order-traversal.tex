\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
\SetKwData{MyTrue}{True}
%%
%% input/output
\Input{A binary tree $T$ on $n > 0$ vertices.}
\Output{A list of the vertices of $T$ in in-order.}
\BlankLine
%%
%% algorithm body
$L \assign [\,]$\;
$S \assign$ empty stack\;
$v \assign$ root of $T$\;
\While{$\MyTrue$}{
  \eIf{\rm $v \neq \texttt{NULL}$}{
    $\push(S, v)$\;
    $v \assign$ left-child of $v$\;
  }{
    \If{$\length(S) = 0$}{
      exit the loop\;
    }
    $v \assign \pop(S)$\;
    $\append(L, v)$\;
    $v \assign$ right-child of $v$\;
  }
}
\Return $L$\;
