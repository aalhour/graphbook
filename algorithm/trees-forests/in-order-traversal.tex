\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
%%
%% input/output
\Input{A binary tree $T$ on $n > 0$ vertices.}
\Output{A list of the vertices of $T$ in in-order.}
\BlankLine
%%
%% algorithm body
$L \assign [\,]$\;
$S \assign$ empty stack\;
$r \assign$ root of $T$\;
$\push(S, r)$\;
\While{$\length(S) > 0$}{
  $v \assign \pop(S)$\;
  \If{\rm $v$ has a left-child}{
    $u \assign$ left-child of $v$\;
    $\push(S, u)$\;
  }
  $\append(L, v)$\;
  \If{\rm $v$ has a right-child}{
    $u \assign$ right-child of $v$\;
    $\push(S, u)$\;
  }
}
\Return $L$\;
