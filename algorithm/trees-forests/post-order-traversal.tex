\DontPrintSemicolon
\SetAlgoNoLine
%%
%% input
\KwIn{An ordered tree $T$ on $n > 0$ vertices.}
%%
%% output
\KwOut{A list of the vertices of $T$ in post-order.}
\BlankLine
%%
%% algorithm body
$L \assign [\,]$\;
$S \assign$ empty stack\;
$r \assign$ root of $T$\;
$\push(S, r)$\;
\While{$\length(S) > 0$}{
  \If{\rm $\Top(S)$ is unmarked}{
    mark $\Top(S)$\;
    $[u_1, u_2, \dots, u_k] \assign$ ordering of children of $\Top(S)$\;
    \For{$i \assign k, k-1, \dots, 1$}{
      $\push(S, u_i)$\;
    }
  }
  \Else{
    $v \assign \pop(S)$\;
    $\append(L, v)$\;
  }
}
\Return $L$\;
