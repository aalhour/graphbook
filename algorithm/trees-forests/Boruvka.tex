\dontprintsemicolon
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
\SetKwData{NULL}{NULL}
%%
%% input/output
\Input{A weighted connected graph $G = (V, E)$ with weight function
  $w$. All the weights of $G$ are distinct.}
\Output{A minimum spanning tree $T$ of $G$.}
\BlankLine
%%
%% algorithm body
$n \assign |V|$\;
$T \assign \overline{K}_n$\;
\While{$|E(T)| < n - 1$}{
  \For{\rm each component $T'$ of $T$}{
    $e' \assign$ edge of minimum weight that leaves $T'$\;
    $E(T) \assign E(T) \cup e'$\;
  }
}
\Return $T$\;
