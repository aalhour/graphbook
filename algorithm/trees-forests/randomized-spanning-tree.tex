\dontprintsemicolon
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
%%
%% input/output
\Input{A connected graph $G$.}
\Output{A spanning tree of $G$.}
\BlankLine
%%
%% algorithm body
$T \assign G$\;
\While{\rm $T$ is not a tree}{
  $e \assign$ random edge of $T$\;
  \If{\rm $T - e$ is connected}{
    $T \assign T - e$\;
  }
}
\Return $T$\;
