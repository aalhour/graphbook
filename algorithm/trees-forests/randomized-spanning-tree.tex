\DontPrintSemicolon
\SetAlgoNoLine
%%
%% input
\KwIn{A connected graph $G$.}
%%
%% output
\KwOut{A spanning tree of $G$.}
\BlankLine
%%
%% algorithm body
$T \assign G$\;
\While{\rm $T$ is not a tree}{
  $e \assign$ random edge of $T$\;
  \If{\rm $T - e$ is connected}{
    $T \assign T - e$\;
  }
}
\Return $T$\;
