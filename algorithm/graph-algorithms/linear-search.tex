\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwData{MyFalse}{False}
\SetKwData{MyTrue}{True}
%%
%% input
\KwIn{A nonempty list $L$ of vertices or edges. An object $E$ for
  which we want to search in $L$.}
%%
%% output
\KwOut{\MyTrue if $E$ is in $L$; \MyFalse otherwise.}
\BlankLine
%%
%% algorithm body
\For{\rm each $e \in L$}{
  \If{$E = e$}{
    \Return \MyTrue\;
  }
}
\Return \MyFalse\;
