\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
%%
%% input/output
\Input{Two integers $a > 0$ and $b \geq 0$ with $a \geq b$.}
\Output{The greatest common divisor of $a$ and $b$.}
\BlankLine
%%
%% algorithm body
$x \assign a$\;
$y \assign b$\;
\While{$y \neq 0$}{
  $r \assign x \bmod y$\;
  $x \assign y$\;
  $y \assign r$\;
}
\Return $x$
