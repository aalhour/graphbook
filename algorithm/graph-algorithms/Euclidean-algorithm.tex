\DontPrintSemicolon
\SetAlgoNoLine
%%
%% input
\KwIn{Two integers $a > 0$ and $b \geq 0$ with $a \geq b$.}
%%
%% output
\KwOut{The greatest common divisor of $a$ and $b$.}
\BlankLine
%%
%% algorithm body
$x \assign a$\;
$y \assign b$\;
\While{$y \neq 0$}{
  $r \assign x \bmod y$\;
  $x \assign y$\;
  $y \assign r$\;
}
\Return $x$
