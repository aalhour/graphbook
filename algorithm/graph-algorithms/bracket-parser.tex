\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
\SetKwData{Left}{left}
%%
%% input/output
\Input{A nonempty string $S$ of characters.}
\Output{A list $L$ of positive integers indicating how the brackets
  match. If the brackets are not balanced, return the empty string
  $\varepsilon$.}
\BlankLine
%%
%% algorithm body
$L \assign [\,]$\;
$T \assign \text{empty stack}$\;
$c \assign 1$\;
$n \assign |S|$\;
\For{$i \assign 0, 1, \dots, n$}{
  \uIf{$S[i+1]$ \rm{is a left bracket}}{
    \text{append $c$ to $L$}\;
    \text{push $(S[i+1],\, c)$ onto $T$}\;
    $c \assign c + 1$\;
  }
  \uIf{$S[i+1]$ \rm{is a right bracket}}{
    \uIf{$T$ \rm{is empty}}{
      \Return $\varepsilon$\;
    }
    $(\Left,\, d) \assign \text{pop } T$\;
    \eIf{\rm{\Left matches $S[i+1]$}}{
      \text{append $d$ to $L$}\;
    }{
      \Return $\varepsilon$\;
    }
  }
}
\uIf{$T$ \rm{is empty}}{
  \Return $L$\;
}
\Return $\varepsilon$\;
