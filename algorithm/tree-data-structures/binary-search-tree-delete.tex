\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
%%
%% input/output
\Input{A nonempty binary search tree $T$ and a vertex $x \in V(T)$ to
  be removed from $T$.}
\Output{The same BST $T$ but without $x$.}
\BlankLine
%%
%% algorithm body
$u \assign \texttt{NULL}$\;
$v \assign \texttt{NULL}$\;
\eIf{\rm $\leftChild[x] \neq \texttt{NULL}$ or $\rightChild[x] \neq \texttt{NULL}$}{
  $v \assign x$\;
}{
  $v \assign$ successor of $x$\;
}
\eIf{\rm $\leftChild[v] \neq \texttt{NULL}$}{
  $u \assign \leftChild[v]$\;
}{
  $u \assign \rightChild[v]$\;
}
\If{\rm $u \neq \texttt{NULL}$}{
  $\parent[u] \assign \parent[v]$\;
}
\eIf{\rm $\parent[v] = \texttt{NULL}$}{
  $\rootElem[T] \assign u$\;
}{
  \eIf{\rm $v = \leftChild[\parent[v]]$}{
    $\leftChild[\parent[v]] \assign u$\;
  }{
    $\rightChild[\parent[v]] \assign u$\;
  }
}
\If{$v \neq x$}{
  $\kappa_x \assign \kappa_v$\;
  copy $v$'s auxilary data into $x$\;
}
