\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
%%
%% input/output
\Input{A nonempty binary search tree $T$.}
\Output{A vertex of $T$ with minimum key.}
\BlankLine
%%
%% algorithm body
$v \assign$ root of $T$\;
\While{\rm $v$ has a left subtree}{
  $v \assign$ root of left subtree of $v$\;
}
\Return $v$\;
