\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwData{NULL}{\footnotesize{NULL}}
%%
%% input
\KwIn{A nonempty binary search tree $T$.}
%%
%% output
\KwOut{A vertex of $T$ with minimum key.}
\BlankLine
%%
%% algorithm body
$v \assign$ root of $T$\;
\While{$\leftChild[v] \neq \NULL$}{
  $v \assign \leftChild[v]$\;
}
\Return $v$\;
