\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwData{NULL}{\footnotesize{NULL}}
%%
%% input
\KwIn{A binary search tree $T$ and a vertex $x$ to be inserted into $T$.}
%%
%% output
\KwOut{The same BST $T$ but augmeneted with $x$.}
\BlankLine
%%
%% algorithm body
$u \assign \NULL$\;
$v \assign$ root of $T$\;
\While{$v \neq \NULL$}{
  $u \assign v$\;
  \If{$\kappa_x < \kappa_v$}{
    $v \assign \leftChild[v]$\;
  }
  \Else{
    $v \assign \rightChild[v]$\;
  }
}
$\parent[x] \assign u$\;
\If{$u = \NULL$}{
  $\rootElem[T] \assign x$\;
}
\Else{
  \If{$\kappa_x < \kappa_u$}{
    $\leftChild[u] \assign x$\;
  }
  \Else{
    $\rightChild[u] \assign x$\;
  }
}
