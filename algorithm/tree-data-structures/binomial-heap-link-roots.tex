\DontPrintSemicolon
\SetAlgoNoLine
%%
%% input
\KwIn{Two copies of $B_{k-1}$, one rooted at $u$ and the other at $v$.}
%%
%% output
\KwOut{The respective roots of two copies of $B_{k-1}$ linked, with
  one root becoming the parent of the other.}
\BlankLine
%%
%% algorithm body
$\parent[u] \assign v$\;
$\sibling[u] \assign \child[v]$\;
$\child[v] \assign u$\;
$\degree[v] \assign \degree[v] + 1$\;
