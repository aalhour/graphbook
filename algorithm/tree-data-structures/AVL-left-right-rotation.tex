\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
%%
%% input/output
\Input{Three vertices $x,y,z$ of an augmented AVL tree $T_v$, where
  $z$ is the first height-unbalanced vertex in the path from $v$ up to
  the root of $T_v$. The roots of the left and right subtrees of $x$
  are denoted $T_1$ and $T_2$, respectively.}
\Output{A left-right double rotation to height-balance the subtree
  rooted at $z$.}
\BlankLine
%%
%% algorithm body
$\leftChild[\parent[z]] \assign x$\;
$\parent[x] \assign \parent[z]$\;
$\parent[z] \assign x$\;
$\rightChild[x] \assign z$\;
$\leftChild[z] \assign \rootElem[T_2]$\;
$\parent[T_2] \assign z$\;
$\leftChild[x] \assign y$\;
$\parent[y] \assign x$\;
$\rightChild[y] \assign \rootElem[T_1]$\;
$\parent[\rootElem[T_1]] \assign y$\;
