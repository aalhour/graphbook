\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}
%%
%% input/output
\Input{A positive integer $n$.}
\Output{The friendship graph $F_n$.}
\BlankLine
%%
%% algorithm body
\If{$n = 1$}{
  \Return $C_3$\;
}
$G \assign$ null graph\;
$N \assign 2n + 1$\;
\For{$i \assign 0, 1, \dots, N-3$}{
  \eIf{\rm $i$ is odd}{
    add edges $(i,\, i+1)$ and $(i,\, N-1)$ to $G$\;
  }{
    add edge $(i, N-1)$ to $G$\;
  }
}
add edges $(N-2,\, 0)$ and $(N-2,\, N-1)$ to $G$\;
\Return $E$\;
